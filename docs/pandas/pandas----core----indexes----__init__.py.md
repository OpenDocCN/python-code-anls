# 定义一个名为 `calculate_area` 的函数，接收两个参数：`length`（长度）和 `width`（宽度）
def calculate_area(length, width):
    # 计算面积，长度乘以宽度，并返回结果
    area = length * width
    return area

# 初始化一个名为 `length` 的变量，赋值为 5
length = 5
# 初始化一个名为 `width` 的变量，赋值为 10
width = 10

# 调用 `calculate_area` 函数，传入 `length` 和 `width` 作为参数，计算面积并将结果保存到 `result` 变量中
result = calculate_area(length, width)

# 打印输出计算得到的面积结果
print(result)
