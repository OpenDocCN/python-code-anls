# 定义一个名为 `sum_values` 的函数，接收一个参数 `numbers`
def sum_values(numbers):
    # 初始化一个变量 `total`，用于存储累加结果，初始值为 0
    total = 0
    # 使用 `for` 循环遍历参数 `numbers` 中的每一个元素
    for num in numbers:
        # 将当前元素 `num` 加到 `total` 中
        total += num
    # 返回累加结果 `total`
    return total
