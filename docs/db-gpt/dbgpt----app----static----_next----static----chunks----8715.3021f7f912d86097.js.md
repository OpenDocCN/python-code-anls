# 定义一个名为 "calculate_factorial" 的函数，接受一个整数参数 n
def calculate_factorial(n):
    # 初始化变量 "result" 为 1，用于存储阶乘的计算结果
    result = 1
    # 使用 for 循环计算阶乘，从 1 到 n（包含 n）
    for i in range(1, n + 1):
        # 将 result 乘以当前循环的值 i，更新 result 的值
        result *= i
    # 返回计算得到的阶乘结果
    return result
