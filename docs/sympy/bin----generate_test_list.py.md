# Importing necessary modules
from __future__ import print_function  # Using print function compatibility across Python versions

from glob import glob  # Importing glob module for file path matching


def get_paths(level=15):
    """
    Generates a set of paths for testfiles searching.

    Examples
    ========

    >>> get_paths(2)
    ['sympy/test_*.py', 'sympy/*/test_*.py', 'sympy/*/*/test_*.py']
    >>> get_paths(6)
    ['sympy/test_*.py', 'sympy/*/test_*.py', 'sympy/*/*/test_*.py',
    'sympy/*/*/*/test_*.py', 'sympy/*/*/*/*/test_*.py',
    'sympy/*/*/*/*/*/test_*.py', 'sympy/*/*/*/*/*/*/test_*.py']

    """
    # Initialize wildcards list with root directory
    wildcards = ["/"]
    # Generate nested wildcards up to specified level
    for i in range(level):
        wildcards.append(wildcards[-1] + "*/")
    # Create paths based on the 'sympy' root and generated wildcards
    p = ["sympy" + x + "test_*.py" for x in wildcards]
    return p

def generate_test_list():
    # Initialize an empty list for storing paths
    g = []
    # Iterate over paths generated by get_paths()
    for x in get_paths():
        # Extend list g with file paths matching the generated pattern
        g.extend(glob(x))
    # Extract module paths from file paths, remove duplicates, and sort
    g = [".".join(x.split("/")[:-1]) for x in g]
    g = list(set(g))
    g.sort()
    return g

if __name__ == '__main__':
    # Generate and print the list of test module paths
    g = generate_test_list()
    print("tests = [")
    for x in g:
        print("    '%s'," % x)
    print("]")
