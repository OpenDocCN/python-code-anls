"""
===========================
Plots with different scales
===========================

Two plots on the same Axes with different left and right scales.

The trick is to use *two different Axes* that share the same *x* axis.
You can use separate `matplotlib.ticker` formatters and locators as
desired since the two Axes are independent.

Such Axes are generated by calling the `.Axes.twinx` method. Likewise,
`.Axes.twiny` is available to generate Axes that share a *y* axis but
have different top and bottom scales.
"""
# 导入matplotlib.pyplot库，并简称为plt
import matplotlib.pyplot as plt
# 导入numpy库，并简称为np
import numpy as np

# 创建一些模拟数据
t = np.arange(0.01, 10.0, 0.01)
data1 = np.exp(t)
data2 = np.sin(2 * np.pi * t)

# 创建一个图形窗口和一个子图
fig, ax1 = plt.subplots()

# 设定第一个子图的x轴标签
ax1.set_xlabel('time (s)')
# 设定第一个子图的y轴标签和颜色
ax1.set_ylabel('exp', color='tab:red')
# 在第一个子图上绘制数据data1，颜色为红色
ax1.plot(t, data1, color='tab:red')
# 设定第一个子图的y轴刻度标签颜色为红色
ax1.tick_params(axis='y', labelcolor='tab:red')

# 使用ax1的twinx()方法创建一个共享x轴的第二个子图
ax2 = ax1.twinx()

# 设定第二个子图的y轴标签和颜色
ax2.set_ylabel('sin', color='tab:blue')
# 在第二个子图上绘制数据data2，颜色为蓝色
ax2.plot(t, data2, color='tab:blue')
# 设定第二个子图的y轴刻度标签颜色为蓝色
ax2.tick_params(axis='y', labelcolor='tab:blue')

# 调整图形布局，避免右边的y轴标签被裁剪
fig.tight_layout()
# 显示图形
plt.show()
