# 导入需要的库
import matplotlib.pyplot as plt  # 导入绘图库matplotlib
import numpy as np  # 导入数值计算库numpy

import matplotlib.tri as tri  # 导入三角形库matplotlib.tri

# %%
# 创建一个未指定三角形的三角形网格（Triangulation），会进行点的Delaunay三角剖分。

# 首先创建点的x和y坐标。
n_angles = 36  # 角度数
n_radii = 8  # 半径数
min_radius = 0.25  # 最小半径
radii = np.linspace(min_radius, 0.95, n_radii)  # 在最小半径到0.95之间生成半径数组

angles = np.linspace(0, 2 * np.pi, n_angles, endpoint=False)  # 在0到2π之间生成角度数组，不包括终点
angles = np.repeat(angles[..., np.newaxis], n_radii, axis=1)  # 沿着第二个轴（列方向）重复角度数组
angles[:, 1::2] += np.pi / n_angles  # 对于偶数列的角度数组增加π/n_angles

x = (radii * np.cos(angles)).flatten()  # 计算x坐标
y = (radii * np.sin(angles)).flatten()  # 计算y坐标

# 创建三角形网格，没有指定三角形，因此进行Delaunay三角剖分。
triang = tri.Triangulation(x, y)

# 掩盖不想要的三角形。
triang.set_mask(np.hypot(x[triang.triangles].mean(axis=1),
                         y[triang.triangles].mean(axis=1))

ax1.triplot(triang, 'bo-', lw=1)  # 绘制三角形网格的轮廓
