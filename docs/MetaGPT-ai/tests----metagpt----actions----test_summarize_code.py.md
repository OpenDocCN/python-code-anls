GAME_PY = """
## game.py
import pygame
from snake import Snake
from food import Food

class Game:
    def __init__(self):
        self.score = 0
        self.level = 1
        self.snake = Snake()
        self.food = Food()

    def start_game(self):
        pygame.init()
        self.initialize_game()
        self.game_loop()

    def initialize_game(self):
        self.score = 0
        self.level = 1
        self.snake.reset()
        self.food.generate()

    def game_loop(self):
        game_over = False

        while not game_over:
            self.update()
            self.draw()
            self.handle_events()
            self.check_collision()
            self.increase_score()
            self.increase_level()

            if self.snake.is_collision():
                game_over = True
                self.game_over()

    def update(self):
        self.snake.move()

    def draw(self):
        self.snake.draw()
        self.food.draw()

    def handle_events(self):
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                pygame.quit()
                quit()
            elif event.type == pygame.KEYDOWN:
                if event.key == pygame.K_UP:
                    self.snake.change_direction("UP")
                elif event.key == pygame.K_DOWN:
                    self.snake.change_direction("DOWN")
                elif event.key == pygame.K_LEFT:
                    self.snake.change_direction("LEFT")
                elif event.key == pygame.K_RIGHT:
                    self.snake.change_direction("RIGHT")

    def check_collision(self):
        if self.snake.get_head() == self.food.get_position():
            self.snake.grow()
            self.food.generate()

    def increase_score(self):
        self.score += 1

    def increase_level(self):
        if self.score % 10 == 0:
            self.level += 1

    def game_over(self):
        print("Game Over")
        self.initialize_game()

"""

SNAKE_PY = """
## snake.py
import pygame

class Snake:
    def __init__(self):
        self.body = [(0, 0)]
        self.direction = (1, 0)

    def move(self):
        head = self.body[0]
        dx, dy = self.direction
        new_head = (head[0] + dx, head[1] + dy)
        self.body.insert(0, new_head)
        self.body.pop()

    def change_direction(self, direction):
        if direction == "UP":
            self.direction = (0, -1)
        elif direction == "DOWN":
            self.direction = (0, 1)
        elif direction == "LEFT":
            self.direction = (-1, 0)
        elif direction == "RIGHT":
            self.direction = (1, 0)

    def grow(self):
        tail = self.body[-1]
        dx, dy = self.direction
        new_tail = (tail[0] - dx, tail[1] - dy)
        self.body.append(new_tail)

    def get_head(self):
        return self.body[0]

    def get_body(self):
        return self.body[1:]

"""
