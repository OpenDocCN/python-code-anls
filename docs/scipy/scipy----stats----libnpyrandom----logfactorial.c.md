/*
 *  计算 log(k!) 的函数
 */

double logfactorial(int64_t k)
{
    // 预先计算常量 halfln2pi = log(sqrt(2*pi))
    const double halfln2pi = 0.9189385332046728;

    // 如果 k 小于预先计算的 logfact 数组的大小，则直接使用查找表中的值
    if (k < (int64_t) (sizeof(logfact)/sizeof(logfact[0]))) {
        /* 使用查找表中的预先计算结果 */
        return logfact[k];
    }

    /*
     *  使用 Stirling 级数展开，截断至 1/k**3 项。
     *  在 Python 实现中，对于 k 最大为 10000000，结果与最佳 64 位浮点值相差不超过 2 ULP。
     */
    return (k + 0.5)*log(k) - k + (halfln2pi + (1.0/k)*(1/12.0 - 1/(360.0*k*k)));
}
