# 定义一个名为 sum_numbers 的函数，接收一个整数列表作为参数
def sum_numbers(numbers):
    # 初始化一个变量 total，用于存储总和，初始值为 0
    total = 0
    # 遍历列表中的每个数字
    for num in numbers:
        # 将当前数字加到 total 上
        total += num
    # 返回计算得到的总和
    return total
