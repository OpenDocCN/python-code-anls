#include <stdio.h>
#include <tvm_runtime.h>
#include <tvmgen_rec.h>

#include "uart.h"

// Header files generated by convert_image.py
#include "inputs.h"
#include "outputs.h"

// 定义字符字典
int char_dict_nums = 97;
// 初始化 UART
uart_init();
// 打印信息
printf("Starting ocr rec inference\n");
// 定义输出结构体
struct tvmgen_rec_outputs rec_outputs = {
    .output = output,
};
// 定义输入结构体
struct tvmgen_rec_inputs rec_inputs = {
    .x = input,
};

// 运行模型
tvmgen_rec_run(&rec_inputs, &rec_outputs);

// 后处理
// 计算字符数量
int char_nums = output_len / char_dict_nums;
// 初始化变量
int last_index = 0;
float score = 0.f;
int count = 0;

// 打印文本信息
printf("text: ");
// 遍历字符数量
for (int i = 0; i < char_nums; i++) {
    // 初始化变量
    int argmax_idx = 0;
    float max_value = 0.0f;
    // 遍历字符字典
    for (int j = 0; j < char_dict_nums; j++){
        // 找到最大概率的字符索引
        if (output[i * char_dict_nums + j] > max_value){
            max_value = output[i * char_dict_nums + j];
            argmax_idx = j;
        }
    }
    // 如果 argmax_idx 大于 0 并且不满足条件 (!(i > 0 && argmax_idx == last_index))，则执行以下代码块
    if (argmax_idx > 0 && (!(i > 0 && argmax_idx == last_index))) {
      // 分数增加最大值
      score += max_value;
      // 计数增加1
      count += 1;
      // 打印 argmax_idx、max_value 和对应的字符
      // printf("%d，%f，%c\n", argmax_idx, max_value, dict[argmax_idx]);
      printf("%c", dict[argmax_idx]);
    }
    // 更新 last_index 为 argmax_idx
    last_index = argmax_idx;
  }
  // 计算平均分数
  score /= count;
  // 打印平均分数
  printf(", score: %f\n");
  
  // 当从 UART 收到 "EXITTHESIM" 时，FVP 将关闭
  printf("EXITTHESIM\n");
  // 进入无限循环，程序停在这里
  while (1 == 1)
    ;
  // 返回 0
  return 0;
